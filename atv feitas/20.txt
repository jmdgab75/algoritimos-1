// Gerar um número aleatório entre 1 e 100
const numeroAleatorio = Math.floor(Math.random() * 100) + 1;

// Variável para armazenar a tentativa atual do usuário
let tentativaAtual;

// Variável para contar o número de tentativas
let numeroTentativas = 0;

// Função para solicitar a tentativa do usuário e fornecer dicas
function adivinharNumero() {
  tentativaAtual = parseInt(prompt("Adivinhe o número (entre 1 e 100):"));

  // Verificar se a tentativa é válida
  if (isNaN(tentativaAtual) || tentativaAtual < 1 || tentativaAtual > 100) {
    alert("Por favor, insira um número válido entre 1 e 100.");
    return;
  }

  // Incrementar o número de tentativas
  numeroTentativas++;

  // Verificar se a tentativa é igual ao número aleatório
  if (tentativaAtual === numeroAleatorio) {
    alert(
      `Parabéns! Você acertou o número em ${numeroTentativas} tentativa(s).`
    );
  } else if (tentativaAtual < numeroAleatorio) {
    alert("Tente um número maior.");
    adivinharNumero();
  } else {
    alert("Tente um número menor.");
    adivinharNumero();
  }
}

// Chamar a função para iniciar o jogo
adivinharNumero();
