const numeroAleatorio = Math.floor(Math.random() * 100) + 1;
let tentativas = 0;
let acertou = false;

while (!acertou) {
  const palpite = parseInt(prompt("Digite um número entre 1 e 100:"));

  tentativas++;

  if (palpite === numeroAleatorio) {
    acertou = true;
    alert(`Parabéns! Você acertou o número em ${tentativas} tentativas.`);
  } else if (palpite < numeroAleatorio) {
    alert("O número correto é maior. Tente novamente.");
  } else {
    alert("O número correto é menor. Tente novamente.");
  }
}
