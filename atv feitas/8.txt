// Variável para armazenar o número atual
let numero = 1;

// Laço while para encontrar o primeiro número divisível por 5 e 7
while (true) {
  if (numero % 5 === 0 && numero % 7 === 0) {
    break;
  }
  numero++;
}

// Imprimir o número encontrado
console.log("O primeiro número divisível por 5 e 7 é: " + numero);
